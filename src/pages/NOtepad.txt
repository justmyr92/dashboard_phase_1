-- Create CAMPUS table
CREATE TABLE CAMPUS (
    CAMPUS_ID VARCHAR(50) PRIMARY KEY,
    CAMPUS_NAME VARCHAR(255),
    CAMPUS_DESCRIPTION VARCHAR(255),
    CAMPUS_EMAIL VARCHAR(255),
    CAMPUS_PHONE VARCHAR(20),
    CAMPUS_ADDRESS VARCHAR(255)
);

-- Create CAMPUS EXTENSION table
CREATE TABLE CAMPUS_EXTENSION (
    CAMPUS_EXT_ID VARCHAR(50) PRIMARY KEY,
    CAMPUS_EXT_NAME VARCHAR(255),
    CAMPUS_EXT_DESCRIPTION VARCHAR(255),
    CAMPUS_EXT_EMAIL VARCHAR(255),
    CAMPUS_EXT_PHONE VARCHAR(20),
    CAMPUS_EXT_ADDRESS VARCHAR(255),
    CAMPUS_ID VARCHAR(50) REFERENCES CAMPUS(CAMPUS_ID)
);

-- Create CSDO table
CREATE TABLE CSDO (
    CSDO_ID VARCHAR(50) PRIMARY KEY,
    CSDO_NAME VARCHAR(255),
    CSDO_DESCRIPTION VARCHAR(255),
    CSDO_EMAIL VARCHAR(255),
    CSDO_PHONE VARCHAR(20),
    CSDO_ADDRESS VARCHAR(255)
);

-- Create SDO table
CREATE TABLE SDO (
    SDO_ID VARCHAR(50) PRIMARY KEY,
    SDO_NAME VARCHAR(255),
    SDO_DESCRIPTION VARCHAR(255),
    SDO_EMAIL VARCHAR(255),
    SDO_PHONE VARCHAR(20),
    SDO_ADDRESS VARCHAR(255)
);

-- Create UNIT table
CREATE TABLE UNIT (
    UNIT_ID VARCHAR(50) PRIMARY KEY,
    UNIT_NAME VARCHAR(255),
    UNIT_DESCRIPTION VARCHAR(255),
    UNIT_EMAIL VARCHAR(255),
    UNIT_PHONE VARCHAR(20),
    UNIT_ADDRESS VARCHAR(255),
    CAMPUS_ID VARCHAR(50) REFERENCES CAMPUS(CAMPUS_ID),
    CAMPUS_EXT_ID VARCHAR(50) REFERENCES CAMPUS_EXTENSION(CAMPUS_EXT_ID),
    SDO_ID VARCHAR(50) REFERENCES SDO(SDO_ID),
    SDG_ID VARCHAR(50) REFERENCES SDG(SDG_ID)
);

-- Create SDG table
CREATE TABLE SDG (
    SDG_ID VARCHAR(50) PRIMARY KEY,
    SDG_NO INT,
    SDG_NAME VARCHAR(255),
    SDG_DESCRIPTION VARCHAR(255)
);

-- Create PROGRAMS table
CREATE TABLE PROGRAMS (
    PROGRAM_ID VARCHAR(50) PRIMARY KEY,
    PROGRAM_NAME VARCHAR(255),
    PROGRAM_DESCRIPTION VARCHAR(255),
    CAMPUS_ID VARCHAR(50) REFERENCES CAMPUS(CAMPUS_ID),
    CAMPUS_EXT_ID VARCHAR(50) REFERENCES CAMPUS_EXTENSION(CAMPUS_EXT_ID)
);

-- Create LADDERS table
CREATE TABLE LADDERS (
    LADDER_ID VARCHAR(50) PRIMARY KEY,
    LADDER_NAME VARCHAR(255),
    LADDER_YEAR INT,
    LADDER_DESCRIPTION VARCHAR(255),
    LADDER_STATUS VARCHAR(50),
    LADDER_DATE DATE
);

-- Create CONTRIBUTIONS table
CREATE TABLE CONTRIBUTIONS (
    CONTRIBUTION_ID VARCHAR(50) PRIMARY KEY,
    CONTRIBUTION_NAME VARCHAR(255),
    CONTRIBUTION_VALUE DECIMAL(10,2),
    CONTRIBUTION_DATE DATE,
    CONTRIBUTION_STATUS VARCHAR(50),
    CONTRIBUTION_REPORT VARCHAR(255),
    UNIT_ID VARCHAR(50) REFERENCES UNIT(UNIT_ID),
    SDG_ID VARCHAR(50) REFERENCES SDG(SDG_ID),
    LADDER_ID VARCHAR(50) REFERENCES LADDERS(LADDER_ID)
);

-- Create ENROLLMENT table
CREATE TABLE ENROLLMENT (
    ENROLLMENT_ID VARCHAR(50) PRIMARY KEY,
    ENROLLMENT_SCHOOL_YEAR VARCHAR(20),
    ENROLLMENT_YEAR_LEVEL INT,
    PROGRAM_ID VARCHAR(50) REFERENCES PROGRAMS(PROGRAM_ID)
);

-- Create ENROLLEES table
CREATE TABLE ENROLLEES (
    ENROLLEE_ID VARCHAR(50) PRIMARY KEY,
    ENROLEE_GENDER VARCHAR(10),
    ENROLLEE_NUMBER INT,
    ENROLLMENT_ID VARCHAR(50) REFERENCES ENROLLMENT(ENROLLMENT_ID)
);

-- Create ACCREDITATION table
CREATE TABLE ACCREDITATION (
    ACCREDITATION_ID VARCHAR(50) PRIMARY KEY,
    ACCREDITATION_NAME VARCHAR(255),
    ACCREDITATION_DATE DATE,
    ACCREDITATION_LEVEL VARCHAR(50),
    ACCREDITATION_STATUS VARCHAR(50),
    CAMPUS_ID VARCHAR(50) REFERENCES CAMPUS(CAMPUS_ID)
);

-- Create PERSONNEL table
CREATE TABLE PERSONNEL (
    PERSONNEL_ID VARCHAR(50) PRIMARY KEY,
    PERSONNEL_TYPE VARCHAR(50) REFERENCES YOUR_PERSONNEL_TYPE_TABLE(PERSONNEL_TYPE_ID),
    PERSONNEL_POPULATION INT,
    POPULATION_YEAR INT,
    CAMPUS_ID VARCHAR(50) REFERENCES CAMPUS(CAMPUS_ID)
);

-- Create ANNUAL REPORT table
CREATE TABLE ANNUAL_REPORT (
    ANNUAL_REPORT_ID VARCHAR(50) PRIMARY KEY,
    ANNUAL_REPORT_YEAR INT,
    ANNUAL_REPORT_DESCRIPTION VARCHAR(255),
    ANNUAL_REPORT_FILE VARCHAR(255),
    SDO_ID VARCHAR(50) REFERENCES SDO(SDO_ID)
);
